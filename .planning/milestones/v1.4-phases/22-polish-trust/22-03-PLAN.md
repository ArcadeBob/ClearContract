**UX-03 -- Date timeline urgency coloring:**

In `DateTimeline.tsx`:
1. Add a `getDateUrgency(dateStr: string)` helper function (or import from a shared util if Plan 02 created one):
   ```typescript
   function getDateUrgency(dateStr: string) {
     const now = new Date();
     now.setHours(0, 0, 0, 0);
     const target = new Date(dateStr);
     target.setHours(0, 0, 0, 0);
     const diffMs = target.getTime() - now.getTime();
     const diffDays = Math.round(diffMs / (1000 * 60 * 60 * 24));

     if (diffDays < 0) return { label: `${Math.abs(diffDays)} days ago`, colorClass: 'text-slate-400', isPast: true };
     if (diffDays === 0) return { label: 'Today', colorClass: 'text-red-600', isPast: false };
     if (diffDays <= 7) return { label: `${diffDays} days away`, colorClass: 'text-red-600', isPast: false };
     if (diffDays <= 30) return { label: `${diffDays} days away`, colorClass: 'text-amber-600', isPast: false };
     return { label: `${diffDays} days away`, colorClass: 'text-emerald-600', isPast: false };
   }
   ```
