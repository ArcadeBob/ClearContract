    2. In src/api/analyzeContract.ts, replace the error handling block (lines 45-47) to handle non-JSON responses gracefully:

       Replace:
       ```typescript
       if (!response.ok) {
           const body = await response.json().catch(() => ({ error: 'Unknown error' }));
           throw new Error(body.error || `Analysis failed (HTTP ${response.status})`);
       }
       ```
       With:
       ```typescript
       if (!response.ok) {
           let errorMessage: string;
           const contentType = response.headers.get('content-type') || '';
           if (contentType.includes('application/json')) {
               const body = await response.json().catch(() => null);
               errorMessage = body?.error || `Analysis failed (HTTP ${response.status})`;
           } else {
               const text = await response.text().catch(() => '');
               errorMessage = text.includes('<!DOCTYPE')
                   ? `Server returned HTML instead of JSON (HTTP ${response.status}). Is the API server running?`
                   : `Analysis failed (HTTP ${response.status}): ${text.slice(0, 200)}`;
           }
           throw new Error(errorMessage);
       }
       ```

       This ensures that if the API route is unreachable (returning HTML 404), the user sees a helpful message pointing them to check the API server, instead of "Unknown error".
